val boostActive = Random.nextBoolean()
if (boostActive) {
    // Move active users the top
    searchQuery.queryNode(BOOST_HANDLE) {
        it.functions.add(
            FunctionScore.Weight(
                1000.0,
                filter = UserDoc.isActive.eq(true)
            )
        )
    }
}
